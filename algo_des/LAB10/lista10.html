def max_flow(C, s, t):
        n = len(C) # C is the capacity matrix
        F = [[0] * n for i in range(n)]
        path = bfs(C, F, s, t)
      #  print path
        while path != None:
            flow = min(C[u][v] - F[u][v] for u,v in path)
            for u,v in path:
                F[u][v] += flow
                F[v][u] -= flow
            path = bfs(C, F, s, t)
        return sum(F[s][i] for i in range(n))

def plot_net(flows):
    cmap = []
    labels = {}

    for edge in flows:
        f = flows[edge]['flow']
        w = flows[edge]['weight']
        labels[edge] = str(f) + '/' + str(w)
        if w == f:
            cmap.append('red')
        elif f:
            cmap.append('orange')
        else:
            cmap.append('black')

    nx.draw(
        G,
        pos=pos,
        with_labels=True
        )

    nx.draw_networkx_edge_labels(
        G,
        pos=pos,
        edge_labels=labels,
        )

    nx.draw_networkx_edges(
        G,
        pos=pos,
        edge_color=cmap
    )
    plt.show()
